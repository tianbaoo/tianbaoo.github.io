package main

import (
	"fmt"
)


func printSlice[T any](s []T){
	for _,v := range s{
		fmt.Printf("%v\n",v)
	}
}

func main(){
	// 使用泛型可以传入多种类型不同的参数，进行函数复用
	printSlice[int]([]int{1,2,3,4,5})
	printSlice[float64]([]float64{1.1,1.2,1.3})
	printSlice[string]([]string{"hello","world","!"})

	// 省略显示类型 省略显示[T any]
	printSlice([]int64{5,4,3,2,1})

}

[T any]参数的类型，意思是该函数支持任何T类型

package main

import (
	"fmt"
)

type anySlice[T any][]T

func printSlice[T any](s []T){
	for _,v := range s{
		fmt.Printf("%v\n",v)
	}
}

func main(){

    a := anySlice[int]{1, 2}
	printSlice(a)

	a2 := anySlice[string]{"hello", "world", "!"}
	printSlice(a2)

	a3 := anySlice[float64]{1.1, 1.2, 1.3}
	printSlice(a3)

}

package main

import (
	"fmt"
)

type M[K string, V any] map[K]V   // 这里的K不支持any ,由于底层map不支持,所以使用string


func printMap[K string, V any](s map[K]V){
	for k, v := range s{
		fmt.Printf("%v:%v\n", k, v)
	}
}

func main(){

    m1 := M[string, int]{"key": 1}
	printMap[string, int](m1)

	m2 := M[string, string]{"hello": "world"}
	printMap[string, string](m2)
    
    m3 := M[string, float64]{"float": 1.2}
    // 省略泛型类型
    printMap(m3)

}

package main

import (
	"fmt"
)

type C[T any] chan T

func printC[T any](s chan T){
	// 第一种遍历方式 chan 必须先被close
	for k := range s{
		fmt.Printf("%v\n", k)
	}

	// 第二种遍历方式
	// for{
	// 	if value, ok := <-s; ok{
	// 		//do somthing
	// 		fmt.Printf("%v\n", value)
	// 	}else{
	// 		break;//表示channel已经被关闭，退出循环
	// 	}
	// }
}



func main(){

    c1 := make(C[int], 2)
	c1 <- 1
	c1 <- 2
	// 使用for range遍历chan，必须先close chan
	close(c1)
	printC(c1)

	c2 := make(C[string], 10)
	c2 <- "hello"
	c2 <- "world"
	// 使用for range遍历chan，必须先close chan
	close(c2)
	printC(c2)

}

package main

import "fmt"

type NumStr interface {
	Num | Str
}

type Num interface {
	~int | ~int8 | ~int16| ~int32| ~int64| ~uint| ~ uint8| ~ uint16| ~ uint32| ~uint64| ~ uintptr| ~ float32| ~ float64| ~ complex64| ~ complex128
}
type Str interface {
	string
}

func add[T NumStr](a,b T) T {
	return a + b
}


func main(){

    fmt.Println(add(3,4))
	fmt.Println(add("hello ","world"))
}

上面的 NumStr，新增了类型列表表达式，它是对类型参数进行约束
使用 | 表示取并集
如果传入参数不在集合限制范围内，就会报错

package main

import "fmt"
import "strconv"

type Price int

func (i Price) String() string {
	//  Price自身的一个方法，将自己转成str类型
	return strconv.Itoa(int(i))
}

type Price2 string

func (i Price2) String() string {
	//  Price自身的一个方法，将自己转成str类型
	return string(i)
}

type ShowPrice interface {
	String() string
	~int | ~string
}

// [T ShowPrice]限制了传入的类型只能是ShowPrice内包含的类型
func ShowPriceList[T ShowPrice](s []T) (ret []string) {
	for _, v := range s {
		ret = append(ret, v.String())
	}
	return
}

func main(){
	// 用strconv包转字符串
    fmt.Println(ShowPriceList([]Price{1, 2}))
	// 用string方法转字符串
	fmt.Println(ShowPriceList([]Price2{"a", "b"}))
    
}

package main

import (
	"fmt"
	"strconv"
)

type Price int

func (i Price) String() string {
	//  Price自身的一个方法，将自己转成str类型
	return strconv.Itoa(int(i))
}


type ShowPrice interface {
	String() string
	~int | ~string
}

// [T ShowPrice]限制了传入的类型只能是ShowPrice内包含的类型
func ShowPriceList[T ShowPrice](s []T) (ret []string) {
	for _, v := range s {
		ret = append(ret, v.String())
	}
	return
}

func main(){
	// 用strconv包转字符串
    fmt.Println(ShowPriceList([]Price{1, 2}))
    // 传入浮点参数，就会因为不是约束类型而报错
}

package main

import (
	"fmt"
)

func findFunc[T comparable](a []T, v T) int {
	for i, e := range a {
		if e == v {
			return i
		}
	}
	return -1
}

func main(){
	fmt.Println(findFunc([]int{1, 2, 3, 4, 5, 6}, 5))
	fmt.Println(findFunc([]string{"q", "w", "e", "r"}, "w"))
    
}

comparable 的约束类型支持整数 和字符，自定义结构体，也可以嵌套在自定义约束中
