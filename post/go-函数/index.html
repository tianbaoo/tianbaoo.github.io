package main

import "fmt"

func Add(a int, b int) (c int) {
	// 返回返回值， 有形参名称的, 直接return
	// go 不支持函数重载 不支持默认参数
	c = a + b
	return
}

func main() {
	ret := Add(5, 7)
	fmt.Println(ret)

}
