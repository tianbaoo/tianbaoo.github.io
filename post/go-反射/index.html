package main

import (
	"fmt"
	"reflect"
)

func reflectType(x interface{}) {
	t := reflect.TypeOf(x)
	fmt.Println(t)
}
func main() {
	a := 10
	reflectType(a)  // 输出：int
	b := "str"
	reflectType(b)  // 输出：string
}

package main

import (
	"fmt"
	"reflect"
)

func reflectType(x interface{}) {
	t := reflect.TypeOf(x)
	fmt.Println("name:", t.Name(), " kind:", t.Kind())
}
func main() {
	var a *float64
	var b int
	reflectType(a)  // 输出：name:   kind: ptr
	reflectType(b)  // 输出：name: int  kind: int

	// 结构体类型
	type person struct {
		name string
	}
	p := person{name: "tianbao"}
	reflectType(p)  // 输出：name: person  kind: struct
	type book struct {
		name string
	}
	bk := book{name: "中国近代史"}
	reflectType(bk)  // 输出：name: book  kind: struct
}

package main

import (
	"fmt"
	"reflect"
)

func reflectType(x interface{}) {
	v := reflect.ValueOf(x)
	k := v.Kind()
	switch k {
	case reflect.Int64:
		// v.Int()从反射中获取整型的原始值，然后通过int64()强制类型转换
		fmt.Printf("type is int64, value is %d\n", int64(v.Int()))
	case reflect.String:
		// v.String()从反射中获取整型的原始值，然后通过string()强制类型转换
		fmt.Printf("type is string, value is %v\n", string(v.String()))
	}
}
func main() {
	var a int64
	var b string
	b = "hello"
	reflectType(a)  // 输出：type is int64, value is 0
	reflectType(b)  // 输出：type is string, value is hello
	// 将int类型的原始值转换为reflect.Value类型
	c := reflect.ValueOf(10)
	fmt.Printf("type c :%T\n", c) // type c :reflect.Value
}

package main

import (
	"fmt"
	"reflect"
)

func reflectType(x interface{}) {
	v := reflect.ValueOf(x)
	if v.Kind() == reflect.Int64 {
		v.SetInt(200)  // 修改的是副本，reflect包会引发panic
	}
}

func reflectType2(x interface{}) {
	v := reflect.ValueOf(x)
	// 反射中使用 Elem()方法获取指针对应的值
	if v.Elem().Kind() == reflect.Int64 {
		v.Elem().SetInt(200)
	}
}

func main() {
	var a int64
	//reflectType(a)  // panic: reflect: reflect.Value.SetInt using unaddressable value
	reflectType2(&a)
	fmt.Printf("a:%v\n", a)  // 输出：a:200
}

isNil()和isValid()

  • IsNil()报告v持有的值是否为nil。v持有的值的分类必须是通道、函数、接口、映射、指针、切片之一；否则IsNil函数会导致panic
  • IsValid()返回v是否持有一个值。如果v是Value零值会返回假，此时v除
  • IsValid、String、Kind之外的方法都会导致panic IsNil()常被用于判断指针是否为空；IsValid()常被用于判定返回值是否有效

package main

import (
	"fmt"
	"reflect"
)

func main() {
	// *int类型指针
	var a *int
	fmt.Println("var a *int IsNil:", reflect.ValueOf(a).IsNil())  // var a *int IsNil: true
	b := 10
	a = &b
	fmt.Println("var a *int IsNil:", reflect.ValueOf(a).IsNil())  // var a *int IsNil: false

	// nil值
	fmt.Println("nil IsNil", reflect.ValueOf(nil).IsValid())  // nil IsNil false

	// 实例化一个匿名结构体
	c := struct {}{}
	// 尝试从b结构体中寻找"abc"字段
	fmt.Println("不存在的结构体成员", reflect.ValueOf(c).FieldByName("abc").IsValid())  // 不存在的结构体成员 false
    // 尝试b结构体中寻找"abc"方法
	fmt.Println("不存在的结构体方法", reflect.ValueOf(c).MethodByName("abc").IsValid())  // 不存在的结构体方法 false

	// 尝试从map中寻找一个不存在的键
	d := make(map[string]string)
	fmt.Println("不存在的键", reflect.ValueOf(d).MapIndex(reflect.ValueOf("abc")).IsValid())  // 不存在的键 false

}

package main

import (
	"fmt"
	"reflect"
)

type student struct {
	Name string `json:"name"`
	Age  int    `json:"age"`
}

func newStudent(name string, age int) *student {
	// Go语言的结构体没有构造函数，我们可以自己实现
	// 因为struct是值类型，如果结构体比较复杂的话，值拷贝性能开销会比较大
	// 故此构造函数返回的是结构体指针类型
	return &student{
		Name: name,
		Age:  age,
	}
}
func main() {
	stu1 := *newStudent("tianbao", 18)
	t := reflect.TypeOf(stu1)
	fmt.Printf("Name:%v, Kind:%v\n", t.Name(), t.Kind()) // Name:student, Kind:struct

	// 通过for循环遍历结构体所有字段信息
	for i := 0; i < t.NumField(); i++ {
		filed := t.Field(i)
		fmt.Printf("name:%s, index:%d, type:%v, json tag:%v\n", filed.Name, filed.Index, filed.Type, filed.Tag.Get("json"))
		// name:Name, index:[0], type:string, json tag:name
		// name:Age, index:[1], type:int, json tag:age
	}

	// 通过字段名获取指定结构体字段信息
	if ageField, ok := t.FieldByName("Age"); ok {
		fmt.Printf("name:%s index:%d type:%v json tag:%v\n", ageField.Name, ageField.Index, ageField.Type, ageField.Tag.Get("json"))
	    // name:Age index:[1] type:int json tag:age
	}

}

package main

import (
	"fmt"
	"reflect"
)

type student struct {
	Name string `json:"name"`
	Age  int    `json:"age"`
}

func (s student) SayHello() string {
	msg := "Hello 你好"
	return msg
}

func (s student) Run() string {
	msg := "run run run"
	return msg
}

func newStudent(name string, age int) *student {
	// Go语言的结构体没有构造函数，我们可以自己实现
	// 因为struct是值类型，如果结构体比较复杂的话，值拷贝性能开销会比较大
	// 故此构造函数返回的是结构体指针类型
	return &student{
		Name: name,
		Age:  age,
	}
}
func main() {
	stu1 := *newStudent("tianbao", 18)
	t := reflect.TypeOf(stu1)
	v := reflect.ValueOf(stu1)

	fmt.Println(t.NumField())  // 2
	for i := 0; i < v.NumMethod(); i++ {
		methodType := v.Method(i).Type()
		fmt.Printf("method name:%v, method type:%T\n", t.Method(i).Name, methodType)  // method name:Run, method type:*reflect.rtype
		// method name:Run, method type:*reflect.rtype
		// method name:SayHello, method type:*reflect.rtype
		var args = []reflect.Value{}
		// 通过反射调用方法传递的参数必须是 []reflect.Value 类型
		v.Method(i).Call(args)
	}

}
