package main

import (
	"fmt"
	"strings"
	"unicode/utf8"
)

func main() {
    s := "Hello World世界"
	// 打印变量s的值
	fmt.Println(s)  // 输出：Hello World世界

	// 打印变量s的长度 世界两个字占用6个字节
	fmt.Println(len(s))  // 输出：17

	// fmt.Sprintf字符串格式化拼接  fmt.Printf不会自带换行
	name := "tianbao"
	age := 18
	s1 := fmt.Sprintf("name: %s, age: %d\n", name, age)
	fmt.Printf(s1)  // 输出：name: tianbao, age: 18

	// 字符串分割
	s2 := strings.Split(s, "o")
	s3 := fmt.Sprintf("type: %T, value: %s", s2, s2)
	fmt.Println(s3)  // 输出：type: []string, value: [Hell  W rld世界]

	// 字符串是否包含某个字符
	s4 := strings.Contains(s, "Hello")
	fmt.Println(s4)  // 输出：true
	s5 := strings.Contains(s, "Heo")
	fmt.Println(s5)  // 输出：false

	// 字符串前缀和后缀判断
	s6 := strings.HasPrefix(s, "He")
	fmt.Println(s6)  // 输出：true
	s7 := strings.HasSuffix(s, "HH")
	fmt.Println(s7)  // 输出：false

	// 字符串出现的子位置 Index会输出第一次出现的位置 LastIndex输出最后一次输出的位置
	index := strings.Index(s, "o")
	fmt.Println(index)  // 输出：4
	index2 := strings.LastIndex(s, "o")
	fmt.Println(index2)  // 输出：7

	// 字符串的join操作
	s8 := []string{"a1", "a2", "a3"}
	s9 := strings.Join(s8, "#")
	fmt.Println(s9)  // 输出：a1#a2#a3

	// byte和rune类型
	// byte uint8类型，或者叫 byte 型，代表了ASCII码的一个字符
	// rune rune类型(int32)，代表一个 UTF-8字符
	// 组成每个字符串的元素叫做"字符"，可以通过遍历或者单个获取字符串元素获得字符。 字符用单引号（'）包裹
	// 当需要处理中文、日文或者其他复合字符时，则需要用到rune类型
	for i := 0; i < len(s); i++ { //byte
		fmt.Printf("%v(%c) ", s[i], s[i])
	}
	fmt.Println()
	// 输出：72(H) 101(e) 108(l) 108(l) 111(o) 32( ) 87(W) 111(o) 114(r) 108(l) 100(d) 228(ä) 184(¸)
	for _, r := range s { //rune
		fmt.Printf("%v(%c) ", r, r)
	}
	fmt.Println()
	// 输出：72(H) 101(e) 108(l) 108(l) 111(o) 32( ) 87(W) 111(o) 114(r) 108(l) 100(d) 19990(世) 30028(界)
	// 因为UTF8编码下一个中文汉字由3~4个字节组成，所以我们不能简单的按照字节去遍历一个包含中文的字符串，否则就会出现上面输出中第一行的结果
	// 字符串底层是一个byte数组，所以可以和[]byte类型相互转换
	// 字符串是不能修改的 字符串是由byte字节组成，所以字符串的长度是byte字节的长度
	// rune类型用来表示utf8字符，一个rune字符由一个或多个byte组成

	// 修改字符串
	// 要修改字符串，需要先将其转换成[]rune或[]byte，完成后再转换为string
	// 无论哪种转换，都会重新分配内存，并复制字节数组
	s10 := "hello"
	// 强制类型转换
	bytesS1 := []byte(s10)
	bytesS1[0] = 'H'
	fmt.Println(string((bytesS1)))  // 输出：Hello
	s11 := "世界"
	runeS2 := []rune(s11)
	runeS2[0] = '天'
	fmt.Println(string(runeS2))  // 天界

	// 字符串切片
	substr := s[6:11]
	fmt.Printf("substr:%T\n", substr)  // 输出：substr:string
    fmt.Println(substr)  // 输出：World

	// 获取字符串字节数
	fmt.Println(len(s)) // 输出：17
	// 获取一个字符串的字符数
	s12 := utf8.RuneCountInString(s)
	fmt.Println(s12)  // 输出：13

	// 字符串替换Replace
	s13 := strings.Replace(s, "o", "a", -1)
	fmt.Println(s13)  // 输出：Hella Warld世界

	// 大小写全字符替换
	s14 := strings.ToLower(s)
	fmt.Println(s14)  // 输出：hello world世界
	s15 := strings.ToUpper(s)
	fmt.Println(s15)  // 输出：HELLO WORLD世界
	
	// 格式化时间
	tm := time.Now()
	rtime := tm.Format("2006-01-02 15:04:05.000")
	fmt.Println(rtime)

}
