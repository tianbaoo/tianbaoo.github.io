go环境搭建

wget https://studygolang.com/dl/golang/go1.17.4.linux-amd64.tar.gz
sudo tar -xvf go1.17.4.linux-amd64.tar.gz
sudo mv go /usr/local
# 配置环境变量
vim ~/.bashrc
写入
export GOROOT=/usr/local/go #编译器路径
export GOPATH=$HOME/go #项目路径
export PATH=$GOPATH/bin:$GOROOT/bin:$PATH
保存后source ~/.bashrc
go env -w GOPROXY=https://goproxy.cn,direct # 代理
